اللہ خَمْدُ لِلَّهِ عَزَّ وَجَلْ اِتْحَاجَ قَرْآنَ وَسَنَتْ كَيْنَ عَالِكِيْنَ فِیْرِ سِیَانِ تَحْرِکِكَ وَعَوْرَتِ
اسلامی، مَکْكُبَ كَيْنَ وَعَوْرَتْ اور سُنْتَوْلِ كَيْنَ خَدْمَتْ كَيْنَ فَقَدِسْ جِزْءِیْ كَيْنَ تَحْتْ عَلَمْ وَیْنْ كَوْ
وُزْیَاجَیْنْ عَامْ كَرْنْتَ كَا اِرَادَهْ رَکْحَتْیْ ہے — ان اُمُورَکَوْکَسْ وَخُوْلِیْ سَرَانْجَامْ وَیْیْتَ کَهْ
لَہْ مَتْعَدُ وَمِجَالِسْ كَاَقِیَامْ عَمْلْ فِیْنْ لَا یا کِیَابْ — اِنْکِیْ مِجَالِسْ فِیْنْ — اِکِیْکَ مُجَالِسْ " " خَصُوصِیْ
اسلامی بھائی" " " یحیْ — وَعَوْرَتْ اسلامی کَهْ کَدْ نَیْ مَاعُولْ فِیْنْ گَوْ کُلْکَهْ — ہِبِرْ — اور نَایِبْ
اسلامی بھائیوولْ کَوْ خَصُوصِی اسلامی بھائی کِیَاجَا تَابْ — اللَّهُ عَزَّ وَجَلْ اور اسْ کَهْ رَسُولْ
صَلْیْ اللَّهُ تَعَالِیْ عَلِیْهِ وَالْهِ وَسَلْمَ کَهْ فَعْلْ وَکَرْمَ اور امِیرَ اِبْسَنْتَ، بَانِیْ وَعَوْرَتْ اسلامی دَامْتَ
بَرْکَانْهُمْ الْعَالِیْهْ کَيْنَ لَکَاهْ فِیْضْ اِثْرْ — یُجَالِسْ خَصُوصِی اسلامی بھائیوولْ فِیْنْ مَکْکُبَ كَيْنَ وَعَوْرَتْ
عَامْ کَرْنْتَ اور اُنْکِیْنْ مَعَاشْرْ — کَا بَا کَرْ وَاَرْفَرْ وَبَنَا نَیْ فِیْنْ مَصْرُوفْ عَمْلْ — اِنْکِیْ سَلْسْ
کَيْنَ اَکِیْکَ خَصُوصِی اسلامی بھائیوولْ کَهْ کَنْتَ وَقَدْ فَوْقَ مُنْعَقَدْ کَنْتَ جَانْتَ وَالْ
12 رَوْرَ وَدْ نَیْ تَرْفِیْکَ کُوْرَسْ مَحْیْ فِیْنْ — اِنْکِیْنْ فَرْضْ عَلِومْ سَکْحَا نَیْ کَهْ عَلَاوَهْ ان
کَيْنَ اَخْلَاقِیْ تَرْبِیْتْ مَحْیْ کَيْنَ جَانْتَ — اِبْ کَنْتَ بِوْنْتَ وَالْ مُتْعَدُ وَدْ نَیْ تَرْفِیْکَ کُوْرَسْ
فِیْنْ گَوْ کُلْکَهْ — ہِبِرْ — اور نَایِبْ اسلامی بھائیوولْ کَيْ عَلَمْ وَیْنْ — وَمُجْلِیْ اور فَلْحْیْ لَکَاهْ کَيْنَ وَجْ
— " مُجَالِسْ خَصُوصِی اسلامی بھائی" — بَانِیْ مُشَاوَرْتْ — اِنْ کِیْکَ یُبَیْضِ کَرْ گَوْ کُلْکَهْ
بِیْرَ — اسلامی بھائیوولْ کَهْ کَلْحْیْ لَظَامْ کُوْمَرْ یْدْ — ہِبِرْ — اور مَنْظَمْ بِنَا نَیْ کَهْ کَنْتَ
اَشْارُوولْ کَيْ زَبَانْ یِرْ مُشْتَلْ وَیْیْ کَتَابْ فِیْنْ فَرَابْمْ کَرْنْتَ کَيْنَ اَشْدْ ضَرُورْتْ — تَا کَرْ گَوْ کُلْکَهْ
بِیْرَ — اور نَایِبْ اسلامی بھائی دَیْ طَلْقَوْلْ فِیْنْ سَکْحَا نَیْ جَانْتَ وَالْ شَرْیْ احْكَامْ وَ
مَسَائِلْ اور سُنْتَوْلْ کَوْ مَصْرُوفْ خَوْدْ — طُورْ یْ یا وَرَکْحْیْ کِلْکَهْ — وَکَرْ خَصُوصِی اسلامی بھائیوولْ

تک بھی علم دین پھیلا کیں اور آتہیں خوب خوب نہیں سکھا کیں علاوہ ازیں ان کتابوں سے عام مبلغین اسلامی بھائیوں کو بھی اس بات کی رہنمائی لے گی کہ گوگے بہرے اسلامی بھائیوں کو اشاروں کی زبان میں وضو عسل اور نماز کا طریقہ کس طرح سکھایا جا سکتا ہے – الٰہ نہل لِلّٰہ عزیز جلّ ان کتابوں کیلئے تحریف عملی طور پر کو ششیں جاری ہیں بلکہ امیر المومس نے ذات پر کتابیں سے ان کے نام پر کہ ان کا کوئی اشارہ نہیں اتر سے تا دم تحریر نہیں اسلحی بھائیوں کے لیے رسالہ ’’انمول بہرے‘‘ بریل (Braille) لیبگو تڑ میں شائع بھی کیا جا چکا ہے – یہ رسالہ (آپے وضو کریں) بھی انہی کا وشوں کا نتیجہ ہے جو ان شاء اللہ عزیز جلّ وضو کے فرائض و سکن سکھنے کے حوالے سے گوگے بہرے اسلامی بھائی کے لیے نہایت مفید ہے – مجلس خصوصی اسلامی بھائی اس کے علاوہ بھی فرض علوم پر مشتمل و گکر کی کتابیں شائع کرنے کا عزم مُصَمّم رکھتی ہے بلکہ گوگے بہرے اسلامی بھائیوں کو علم دین سکھانے کے ساتھ ساتھ عصری علوم سے روشناں کروانے کے لیے مجلس دار المدین کی معاونت سے ان کے لیے باقاعدہ ایک اسلامی اسکول کھولے کا ارادہ رکھتی ہے – اسی طرح ناپنا اسلامی بھائیوں کیلئے بھی عنقریب جامعہ المدین کا افتتاح کیا جا سکتا ہے جس میں انہیں قرآن وحدیث و فقد کے ساتھ ساتھ و گکر علوم و بینب بھی سکھائے جائیں گے اور مدنی تر بیٹی کو روز قائم کر کے فرض علوم بھی سکھائے جائیں گے – اللہ عزیز جلّ اس کوشش کو اپنی بار گاہ میں قبول فرما تے اور دعوت اسلامی کی تمام مجلس بشمول مجلس خصوصی اسلامی بھائی کو ون پھیپو یں اور رات پھیپو یں ترقی عطا فرما تے اور امیر المومس نے ذات پر کتابیں سے ان کے نام پر کہ ان کا کوئی اشارہ نہیں اتر سے تا دم تحریر نہیں اسلح

# ٹھیک ٹھنڈی: ٹھیک کیلیے

برٹھمرات بعد ٹھنڈہ مغرب آب کے پیالہ ہونے والے دعوت اسلامی کے بھٹنہ و ارسٹنوں بھرے
اجتماع میں رضا نے الٰہی کیلیے اُنہیں اُنہیں بیٹوں کے ساتھ ساری رات شرکت فرمائی ہے۔ سٹنوں
کی تربیت کے لئے مذہبی قا لظ میں عا بشقان رسول کے ساتھ ہر ماہ تین دن سفر اور روزانہ
“تھمر ں یتہ” کے ذریعے مذہبی اِنحا مات کا رسالہ پُر کر کے ہر مذہبی ماہ کی پہلی تاریخ اپنے
پیالے کے ذرتے دار کو چھوڑ کر والے کا معمول بنائی ہے۔

## میرا مذہبی مقصد:

“مجھے اپنی اور ساری دنیا کے لوگوں کی اصلاح کی کوشش
کر نی ہے۔” اِن شاکو اللہ علّٰی - اپنی اصلاح کے لیے “مذہبی اِنحا مات” پر عمل اور ساری دنیا
کے لوگوں کی اصلاح کی کوشش کے لیے “مذہبی قا فلوں” میں سفر کرنا ہے۔ اِن شاکو اللہ علّٰی
